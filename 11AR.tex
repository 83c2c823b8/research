\begin{prop}
	$k$を体，$A$を有限次元$k$代数，$\gldim A <\infty$と仮定する．このとき，
	\[\nu  \coloneq - \Ltensor_A A^\vee \colon D^b(\mod A)\rightarrow D^b(\mod A)\]
	はSerre函手を与える．
\end{prop}
\begin{proof}
	任意の$X,Y\in \D^b(\mod A)$に対して，
	\begin{align*}
		\RHom_A(X,Y\Ltensor_A A^\vee)^\vee &\simeq \left(Y\Ltensor_A \RHom_A(X,A^\vee)\right)^\vee\\
		\simeq (Y\Ltensor_A X^\vee)^\vee &\simeq \RHom_A(Y\Ltensor_A X^\vee) \simeq \RHom_A(Y,(X^\vee)^\vee).
	\end{align*}
	$\H^0$を取れば，示される．
\end{proof}

\begin{defn}

三角圏 $\mathcal{T}$ に対して，その根基（radical）$\mathrm{rad}(\mathcal{T})$を以下のように定める：\\
各対象 $X,Y \in \mathcal{T}$ に対し
\[
\mathrm{rad}_{\mathcal{T}}(X,Y)
=
\left\{
f \in \mathrm{Hom}_{\mathcal{T}}(X,Y)
\;\middle|\;
1_X-gf \text{ が任意の } g:Y\to X \text{ に対して可逆}
\right\}.
\]
によって定まるイデアルである．特に $\mathcal{T}$ が Krull--Schmidt 圏であるとき，$\mathrm{rad}(\mathcal{T})$ は既約でない射全体から生成されるイデアルと一致する．より具体的には，$X,Y$ を直既約対象とすると，
\[
\mathrm{rad}_{\mathcal{T}}(X,Y)
=
\begin{cases}
\Hom_{\T}(X,Y),
& X \not\cong Y,\\[4pt]
\mathrm{rad}\!\left(\End_{\T}(X)\right),
& X \cong Y.
\end{cases}
\]

ここで
\(
\mathrm{rad}\!\left(\mathrm{End}_{\mathcal{T}}(X)\right)
\)
は自己準同型環
\(
\mathrm{End}_{\mathcal{T}}(X)
\)
の Jacobson 根基を表す．また，Auslander--Reiten 理論において，既約射は
\[
\mathrm{rad}_{\mathcal{T}}(X,Y)
\setminus
\mathrm{rad}_{\mathcal{T}}^{2}(X,Y)
\]
の元として特徴付けられる．さらに，Auslander--Reiten quiver の矢印の重複度は
\[
\mathrm{rad}_{\mathcal{T}}(X,Y)
\Big/
\mathrm{rad}_{\mathcal{T}}^{2}(X,Y)
\]
の次元によって与えられる．
\end{defn}
\begin{defn}\cite{Hap88}
	$\C$をKrull-Schmidt圏，$f\in\Hom_{\C}(X,Y)$とする．このとき，\vspace{-3mm}
	\begin{itemize}
		\item[(1)]
			$f\circ g = \id_Y$となるような$g\colon Y\to X$が存在するとき，$f$はsplit epimorphismであるという．
		\item[(2)]
			$f$がsplit epimorphismでなく，任意のsplit epimorphismでない$g\colon Z\to Y$に対して，$g = f\circ h$となるような$h\colon Z\to X$が存在するとき，$f$をright almost splitという．
		\item[(3)]
			任意の$g\in\End(X)$に対して，$f=g\circ f$ならば$g$が同型射がなりたつならば，$f$をright minimalという．
		\item[(4)]
			$f$がright almost splitかつright minimalのとき，$f$をsink morphismという．
	\end{itemize}
\end{defn}

\begin{thm}\cite{Hap88}
	任意の$X\in\ind\T$に対して，以下の完全三角形が存在する：\vspace{-3mm}
	\begin{itemize}
		\item 
		\[
		\begin{tikzcd}
			\Serre(X[-1])\ar[r,"g"]& Y\ar[r,"f"]& X\ar[r] & \Serre(X)
		\end{tikzcd}.
	\]
ただし，$f'$はsink morphismで，$g'$はsource morphismである．
\item
		\[
		\begin{tikzcd}
			X\ar[r,"f'"]& Y'\ar[r,"g'"]& \Serre^{-1}(X[1])\ar[r] & X[1]
		\end{tikzcd}.
	\]
ただし，$f$はsource morphismで，$g$はsink morphismである．
	\end{itemize}
\end{thm}
\begin{proof}
	$X\in\ind\T$を任意にとる．Serre函手の定義により，
		\[\End_{\T}(X)^\vee\simeq \Hom_{\T}(X,\Serre(X)),\]
		また，$(\End_{\T}(X)/\mathrm{rad}(X,X))^\vee \subset (\End_{\T}(X))^\vee$なので，
		$\varphi(\neq 0)\in (\End_{\T}(X)/\mathrm{rad}(X,X))^\vee $をとり，上記の同型で送った射を$h$とする．このとき，
		\[\Serre(X[-1])\xrightarrow{g}Y\xrightarrow{f} X\xrightarrow{h} \Serre(X).\]
		が主張する完全三角形になることを示す．まず，$f$がright almost splitであることを示す．任意の対象$Z\in\T$に対して，以下の列：
		\[\Hom_{\T}(Z,\Serre(X[-1]))\longrightarrow \Hom_{\T}(Z,Y)\xlongrightarrow{f\circ -} \Hom_{\T}(Z,X)\xlongrightarrow{h\circ -} \Hom_{\T}(Z,\Serre(X)).\]
		が完全なので，任意のsplit epimorphismでない$a\in\Hom_{\T}(Z,X)$に対して，$h\circ a = 0$を示せばよい．$a$はsplit epimorphismでなく，$X$が既約なので$a\in\mathrm{rad}(Z,X)$である．
\end{proof}

\begin{lemm}\label{lemm:free-C-action}
	安定性条件$\sigma = (Z,\P)$への複素数の作用は自由である．すなわち以下が成り立つ：\\
	$\xi,\eta\in\CC$としたとき，
	\[\sigma\cdot \xi = \sigma\cdot \eta\Longrightarrow  \xi= \eta.\]
\end{lemm}
\begin{proof}
	$\sigma = (Z,\P)$に対して，$\sigma\cdot (x+\sqrt{-1}y)= \sigma$のとき，$x+\sqrt{-1}y = 0 \quad (x,y\in\RR)$を示せばよい．$ (e^{-\sqrt{-1}\pi(x+\sqrt{-1}y)}Z,\P(\bullet + x))$であるので，任意の$\phi\in\RR$に対して，
	\[\P(\phi + x) = \P(\phi).\]
	が成り立つ，したがって，$x=0$である．また，$e^{\pi y}Z = Z$なので$e^{\pi y} =1$．したがって，$y =0$
\end{proof}

\begin{thm}
	$\overrightarrow{\Delta}$をADE型箙，$\D = D^b(\mod k\overrightarrow{\Delta})$とし，そのSerre函手を$\Serre$
	とする．このとき，$\Serre$に関してGepner型である安定性条件が存在し，$\CC$作用を除いて一意である．
\end{thm}
\begin{proof}
	$\sigma = (Z,\P)$を$\Serre$に関してGepner型であるとすると，$\xi\in\CC$を用いて
\[\Serre\cdot \sigma = \sigma\cdot\xi .\]
したがって，$\Serre^h\cdot\sigma = \sigma\cdot h\xi$となる．また，AR理論より，
	\[\Serre^h \simeq (\tau[1])^h \simeq \tau^h[h] \simeq [h-2].\]
したがって，
	\[\sigma \cdot (h-2)=[h-2]\cdot \sigma = \sigma\cdot h\xi.\]
	(補題\ref{lemm:free-C-action})より，$h-2 = h\xi$，すなわち，$\xi = 1 -\frac{2}{h}$がしたがう．\\
次に中心電荷$Z$が$\CC$作用を除いて一意的であることを示す．
Grothendieck群上で，
\[Z\circ \left[S^{-1}\right] = Ze^{-\sqrt{-1}\pi\xi}.\]
ここで，$\xi = 1 - \frac{2}{h}$であったので，
\[Z\circ \left[\Phi\right] = Ze^{\frac{2\pi\sqrt{-1}}{h}}.\]
したがって，$Z$は固有値$e^{\frac{2\sqrt{-1}\pi}{h}}$の固有空間ベクトルであるので，相異なるものは$\CC$倍の違いのみである．\\
最後に，スライス$\P$の一意性を示す．Dynkin箙の表現の圏ではJordan-H\"olderフィルトレーション，その導来圏はHarder-Narasimhan分解を持つため
\end{proof}
